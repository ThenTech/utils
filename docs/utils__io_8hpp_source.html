395  /**
396  * Read one bit from the bitstream.
397  *
398  * @return The value of the bit.
399  */
400  uint8_t get_bit(void) {
401  const size_t current_start_byte = this->position / 8u;
402 
403  if (current_start_byte >= this->get_size()) {
404  // Prevent reading byte outside of array (-> Valgrind flagged)
405  return 0u;
406  }
407 
408  const size_t bits_taken = this->position % 8;
409  const uint8_t value = this->buffer[current_start_byte];
410 
411  this->position++;
412 
413  return (value & (1 << (7 - bits_taken))) != 0;
414  }
415 
416  /**
417  * Get l bits from the bitstream
418  *
419  * @param [in] l number of bits to read
420  * @return The value of the bits read
421  *
422  * buffer: 0101 1100, position==0
423  * get(4) returns value 5, position==4
424  */
425  uint32_t get(size_t l) {
426  uint32_t value = 0;
427 
428  for (size_t i = 0; i < l; i++) {
429  const uint32_t v = this->get_bit();
430  value |= v << (l - i - 1);
431  }
432 
433  return value;
434  }
