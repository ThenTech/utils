435  /**
436  * Read one bit from the bitstream.
437  *
438  * @return The value of the bit.
439  */
440  uint8_t get_bit(void) {
441  const size_t current_start_byte = this->position / 8u;
442 
443  if (HEDLEY_UNLIKELY(current_start_byte >= this->get_size())) {
444  // Prevent reading byte outside of array (-> Valgrind flagged)
445  return 0u;
446  }
447 
448  const uint_fast32_t bits_taken = this->position % 8;
449  const uint8_t value = this->buffer[current_start_byte];
450 
451  this->position++;
452 
453  return utils::bits::select_one<uint8_t>(value, 8 - bits_taken);
454  }

458  *
459  * @param [in] l number of bits to read
