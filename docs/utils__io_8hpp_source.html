590  *
591  * @return The value of the bit.
592  */
593  uint8_t get_bit(void) {
594  const size_t current_start_byte = this->position / 8u;
595 
596  if (HEDLEY_UNLIKELY(current_start_byte >= this->get_size())) {
597  // Prevent reading byte outside of array (-> Valgrind flagged)
598  return 0u;
599  }

602  const uint8_t value = this->buffer[current_start_byte];
603 
604  this->position++;
