155  bool Function_Aborts(F&& f, Args&& ... args) {
156  static_assert(std::is_invocable_v<F, Args...>, "Function_Aborts: Callable function required.");
157 
158  auto previous_handler = std::signal(SIGABRT, utils::Catch::internal::_function_abort_signal_handler);
159 
160  if (previous_handler == SIG_ERR) {
161  return true;
162  }
163

165

167  std::invoke(std::forward<F>(f), std::forward<Args>(args)...);
168  } else {
