153  /**
154  * \brief Bitwise rotate the given \p value to the left by \p n bits.
155  *
156  * \param value
157  * The value to rotate.
158  * \param n
159  * The amount of bit places to rotate left with.
160  * \return The result will be of type T but with \p bit_length bits used,
161  * other bits exceeding the length will become zeroes.
162  * The value will be shifted to the left, and any bits exceeding
163  * \p bit_length will be appended to the right.
164  */
165  template <
166  class T,
167  size_t bit_length = utils::bits::size_of<T>()
168  > ATTR_MAYBE_UNUSED ATTR_NODISCARD
169  static inline constexpr T rotl(const T value, const int_fast32_t n = 1) {
170  static_assert(std::is_integral_v<T>, "utils::bits::rotl: Integral required.");
171 
172  constexpr T mask_size = bit_length == 64 ? uint64_t(~0ull) : ~(uint64_t(~0) << bit_length);
173  const T mask_data = ~(uint64_t(~0) << n);
174  return ((value << n) & mask_size) | ((value >> (bit_length - n)) & mask_data);
175  }

177  /**
178  * \brief Bitwise rotate the given \p value to the right by \p n bits.
179  *
180  * \param value
181  * The value to rotate.
182  * \param n
183  * The amount of bit places to rotate right with.
184  * \return The result will be of type T but with \p bit_length bits used,
185  * other bits exceeding the length will become zeroes.
186  * The value will be shifted to the right, and any bits shifted out
187  * will be appended to the left starting at \p bit_length.
188  */
189  template <
190  class T,
191  size_t bit_length = utils::bits::size_of<T>()
192  > ATTR_MAYBE_UNUSED ATTR_NODISCARD
193  static inline constexpr T rotr(const T value, const int_fast32_t n = 1) {
194  static_assert(std::is_integral_v<T>, "utils::bits::rotr: Integral required.");
195 
196  const T mask_right = ~(uint64_t(~0) << (bit_length - n));
197  const T mask_data = ~(uint64_t(~0) << n);
198  return ((value >> n) & mask_right) | ((value & mask_data) << (bit_length - n));
199  }

270  /**
271  * @brief Create a signed value of type T
272  * with the given value that is only src_bits long.
273  *
274  * This abuses the fact that right-shifting a signed number
275  * with an MSB of 1, will fill in the new MSBs as 1, making the
276  * original value signed in the new (bigger) size.
277  * @tparam T
278  * Target type (signed).
279  * @param value
280  * The value to make signed in the available bits of T.
281  * @param src_bits
282  * The amount of bits that are used in value.
283  * @return Returns a signed or unsigned number,
284  * depending on whether value[src_bits : 0] was considered signed.
285  */
286  template<
287  class T,
288  size_t Tlen = utils::bits::size_of<T>()
289  > ATTR_MAYBE_UNUSED ATTR_NODISCARD
290  static inline T shift_signed(const size_t value, const size_t src_bits) {
291  #if UTILS_BITS_ASSERT_SHIFT_SIGNED_SIZE
292  // ASSERT that src_bits is not bigger than sizeof(T)
293  ASSERT(src_bits <= Tlen);
294  const size_t bit_length = Tlen - src_bits;
295  return T(value << bit_length) >> bit_length;
296  #else // If src_bits is bigger than sizeof(T), don't shift
297  const uint64_t bit_length = src_bits <= Tlen ? Tlen - src_bits : 0;
298  return T(value << bit_length) >> bit_length;
299  #endif
300  }

321  /**
322  * \brief Determine the amount of bits needed to represent the given T.
323  *
324  * \param value
325  * The T value to check.
326  * \return
327  * Returns the amount of bits required to represent the value if reshifted to size_of<T> bits.
328  */
329  template<class T> ATTR_MAYBE_UNUSED ATTR_NODISCARD
330  static inline uint_fast32_t bits_needed(T value) {
331  static_assert(std::is_integral_v<T>, "utils::bits::bits_needed: Integral required.");
332 
333  uint_fast32_t bits = 0;
334 
335  // 1. Mask value with amount of current bits : (value & ((1 << bits) - 1))
336  // 2. Shift left to size of 16 bits : << (16 - bits)
337  // 3. Cast to int16_t
338  // 4. Shift back to original size
339  // 5. Check if value represented with <bits> bits and re-shifted to 16 bits
340  // equals the starting value.
341  // 6. Repeat until a minimal amount of bits has been found to represent the int16_t.
342  // (between 1 and 16)
343 
344  while (utils::bits::shift_signed<T>(value, ++bits) != value);
345 
346  return bits;
347  }

349  /**
350  * \brief Accumulate the boolean vector to the target type \p T.
351  * The MSB will be the first value in the vector.
352  *
353  * Optional ASSERT to limit size of vector to size of target type.
354  * If `bool_vec.size()` is bigger, the MSBs will be shifted out.
355  *
356  * \tparam T
357  * The target type.
358  * \param bool_vec
359  * The vector to put its contents into a \p T.
360  * \return Returns a `T` where the MSB is the first item in \p bool_vec,
361  * and the LSB is the last one.
362  */
363  template<
364  typename T, typename U = bool,
365  typename = typename std::enable_if_t<std::is_integral_v<T>>,
366  typename = typename std::enable_if_t<std::is_same_v<bool, U>
367  || std::is_integral_v<U>>
368  > ATTR_MAYBE_UNUSED ATTR_NODISCARD
369  static inline T to_binary(const std::vector<U>& bool_vec) {
370  ASSERT(bool_vec.size() <= utils::bits::size_of<T>());
371  return std::accumulate(bool_vec.begin(), bool_vec.end(), T(0),
372  [](T x, T y) { return (x << 1) | (y & 1); });
373  }
374 }
