418  /**
419  * @brief Create a signed value of type T
420  * with the given value that is only src_bits long.
421  *
422  * This abuses the fact that right-shifting a signed number
423  * with an MSB of 1, will fill in the new MSBs as 1, making the
424  * original value signed in the new (bigger) size.
425  * @tparam T
426  * Target type (signed).
427  * @param value
428  * The value to make signed in the available bits of T.
429  * @param src_bits
430  * The amount of bits that are used in value.
431  * @return Returns a signed or unsigned number,
432  * depending on whether value[src_bits : 0] was considered signed.
433  */
434  template<
435  class T,
436  size_t Tlen = utils::bits::size_of<T>()
437  > ATTR_MAYBE_UNUSED ATTR_NODISCARD
438  static inline T shift_signed(const uint64_t value, const size_t src_bits) {
439  #if UTILS_BITS_ASSERT_SHIFT_SIGNED_SIZE
440  // ASSERT that src_bits is not bigger than sizeof(T)
441  ASSERT(src_bits <= Tlen);
442  const size_t bit_length = Tlen - src_bits;
443  return T(value << bit_length) >> bit_length;
444  #else // If src_bits is bigger than sizeof(T), don't shift
445  const uint64_t bit_length = src_bits <= Tlen ? Tlen - src_bits : 0;
446  return T(value << bit_length) >> bit_length;
447  #endif
448  }

469  /**
470  * \brief Determine the amount of bits needed to represent the given T.
471  *
472  * \param value
473  * The T value to check.
474  * \return
475  * Returns the amount of bits required to represent the value if reshifted to size_of<T> bits.
476  */
477  template<class T> ATTR_MAYBE_UNUSED ATTR_NODISCARD
478  static inline uint_fast32_t bits_needed(T value) {
479  static_assert(std::is_integral_v<T>, "utils::bits::bits_needed: Integral required.");
480 
481  uint_fast32_t bits = 0;
482 
483  // 1. Mask value with amount of current bits : (value & ((1 << bits) - 1))
484  // 2. Shift left to size of 16 bits : << (16 - bits)
485  // 3. Cast to int16_t
486  // 4. Shift back to original size
487  // 5. Check if value represented with <bits> bits and re-shifted to 16 bits
488  // equals the starting value.
489  // 6. Repeat until a minimal amount of bits has been found to represent the int16_t.
490  // (between 1 and 16)
491 
492  while (utils::bits::shift_signed<T>(value, ++bits) != value);
493 
494  return bits;
495  }

498  * \brief Accumulate the boolean vector to the target type \p T.
499  * The MSB will be the first value in the vector.
500  *
501  * Optional ASSERT to limit size of vector to size of target type.
502  * If `bool_vec.size()` is bigger, the MSBs will be shifted out.
503  *
504  * \tparam T
505  * The target type.
506  * \param bool_vec
507  * The vector to put its contents into a \p T.
508  * \return Returns a `T` where the MSB is the first item in \p bool_vec,
509  * and the LSB is the last one.
510  */
511  template<
512  typename T, typename U = bool,
513  typename = typename std::enable_if_t<std::is_integral_v<T>>,
514  typename = typename std::enable_if_t<std::is_same_v<bool, U>
515  || std::is_integral_v<U>>
516  > ATTR_MAYBE_UNUSED ATTR_NODISCARD
517  static inline T to_binary(const std::vector<U>& bool_vec) {
518  ASSERT(bool_vec.size() <= utils::bits::size_of<T>());
519  return std::accumulate(bool_vec.begin(), bool_vec.end(), T(0),
520  [](T x, T y) { return (x << 1) | (y & 1); });
521  }
