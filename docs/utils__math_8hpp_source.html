248  *
249  * \param first

252  * The end iterator.

256  typename Iterator,
257  typename DiffType = typename std::iterator_traits<Iterator>::difference_type,
258  typename = typename std::enable_if_t<utils::traits::is_iterator_v<Iterator>>
259  > ATTR_MAYBE_UNUSED ATTR_NODISCARD
260  static double variance(Iterator first, Iterator last) {
261  const DiffType size = std::distance(first, last);
262  ASSERT(size > 1);
263 
264  const double mean = utils::math::stats::mean(first, last);
265  const double sq_sum = std::inner_product(first, last, first, 0.0,
266  [ ](const double& x, const double& y) { return x + y; },
267  [mean](const double& x, const double& y) { return (x - mean) * (y - mean); });
268 
269  return sq_sum / (size - 1);
270  }
