27  static T lexical_cast(const std::string_view buffer) {
28  T out;
29 
30  if (HEDLEY_UNLIKELY(buffer.size() == 0))
31  throw utils::exceptions::CastingException(std::string(buffer), utils::print::type2name(out));
32 
33  std::stringstream cast;
34 
35  if (buffer.size() > 1 && buffer[0] == '0' && buffer[1] != '.') {
36  if (buffer[1] == 'x' || buffer[1] == 'X') {
37  // Hex literal
38  cast << std::hex << buffer;
39  } else if (buffer[1] == 'b' || buffer[1] == 'B') {
40  // Binary literal
41  try {
42  cast << std::stoll(std::string(buffer.substr(2)), nullptr, 2);
43  } catch (std::invalid_argument const&) {
44  throw utils::exceptions::CastingException(std::string(buffer), utils::print::type2name(out));
45  }
46  } else {
47  // Octal literal
48  cast << std::oct << buffer;
49  }
50  } else if (buffer[0] == '#') {
51  // Hex colour string
52  cast << std::hex << buffer.substr(1);
53  } else {
54  // Decimal/float/...
55  cast << buffer;
56  }
57 
58  if (!(cast >> out))
59  throw utils::exceptions::CastingException(std::string(buffer), utils::print::type2name(out));
60 
61  return out;
62  }
