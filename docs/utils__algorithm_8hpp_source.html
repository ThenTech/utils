21  * \brief Check if the given container contains the given item.
22  * The type of item must be something that the container holds.
23  * e.g.
24  * vector<int> <- int
25  * string <- char
26  * and not:
27  * vector<Node> /- int
28  *
29  * \param container
30  * The container to look in.
31  * \param item
32  * The item to look for.
33  * \return Returns an `std:optional<size_t>` holding the offset from
34  * std::begin(container) if the item was found, or nothing if not.
35  */
36  template<
37  typename C, typename T,
38  typename = typename std::enable_if_t<utils::traits::is_container<C>::value>
39  > ATTR_MAYBE_UNUSED ATTR_NODISCARD
40  static inline constexpr utils::traits::found_t contains(const C& container, const T& item) {
41  if constexpr (utils::traits::has_find_v<C, T>) {
42  if (const auto idx = container.find(item); std::end(container) != idx) {
43  return { std::distance(std::begin(container), idx) };
44  }
45  } else {
46  if (const auto idx = std::find(std::begin(container), std::end(container), item);
47  std::end(container) != idx)
48  {
49  return { std::distance(std::begin(container), idx) };
50  }
51  }
52 
53  return std::nullopt;
54  }
