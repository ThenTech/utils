44  * \return Returns a Container instance with the picked items.
45  */

47  typename Container,
48  typename = typename std::enable_if_t<utils::traits::is_iterable_v<Container>>
49  > ATTR_MAYBE_UNUSED ATTR_NODISCARD
50  static auto pick_x_from(size_t amount, const Container& container) {
51  Container picked;
52  picked.reserve(amount);
53 
54  const auto size = std::distance(std::begin(container), std::end(container));
55  if(HEDLEY_UNLIKELY(0 == size)) return picked;
56 
57  using diff_t = typename std::iterator_traits<decltype(std::begin(container))>::difference_type;
58 
59  for (size_t i = 0; i < amount; ++i) {
60  picked.push_back(*std::next(std::begin(container), Random::get<diff_t>( 0, size - 1)));

66  /**
67  * \brief Generate a container (default: vector<T>) with \p amount items,
68  * randomly distributed between \p from and \p to.
69  *
70  * Only fundamental data types are supported (numbers, chars and bools).
71  * These are randomly chosen between between \p from and \p to,
72  * while booleans are chosen with a 1/2 probability.
73  *
74  * To generate booleans with other probabilities, use generate_bool,
75  * to generate a string (instead of vector) when choosing chars,
76  * use generate_string.
77  *
78  * For other data types, the function will not be enabled.
79  *
80  * \param amount
81  * The amount of items to generate.
82  * \param from
83  * The lower bound for the distribution. (default: min T)
84  * \param to
85  * The upper bound for the destribution. (default: max T)
86  * If upper < lower, these are swapped.
87  * \return Returns a new Container containing \p amount items of type T.
88  */
89  template<
90  typename T,
91  typename Container = std::vector<T>,
92  typename = typename std::enable_if_t<utils::traits::is_iterable_v<Container>
93  && ( effolkronium::details::is_supported_number<T>::value
94  || effolkronium::details::is_supported_character<T>::value
95  || std::is_same<T, bool>::value
96  )>
97  > ATTR_MAYBE_UNUSED ATTR_NODISCARD
98  static auto generate_x(const size_t amount,
99  T from = std::numeric_limits<T>::min(),
100  T to = std::numeric_limits<T>::max())
101  {
102  Container picked;
103  picked.reserve(amount);
104 
105  if (HEDLEY_UNLIKELY(to < from)) // Allow range from higher to lower
106  std::swap(from, to);
107 
108  if constexpr (effolkronium::details::is_uniform_real<T>::value) {
109  auto dist = Random::real_dist_t<T> {
110  static_cast<T>(from),
111  static_cast<T>(to)
112  };
113 
114  for (size_t i = 0; i < amount; ++i) {
115  picked.push_back(static_cast<T>(dist(Random::engine())));
116  }
117  } else if constexpr (std::is_same<T, bool>::value) {
118  auto dist = Random::bool_dist_t{ 0.5 };
119 
120  for (size_t i = 0; i < amount; ++i) {
121  picked.push_back(static_cast<T>(dist(Random::engine())));
122  }
123  } else {
124  auto dist = Random::integer_dist_t<int64_t> {
125  static_cast<int64_t>(from),
126  static_cast<int64_t>(to)
127  };
128 
129  for (size_t i = 0; i < amount; ++i) {
130  picked.push_back(static_cast<T>(dist(Random::engine())));
131  }
132  }
133 
134  return picked;
135  }
