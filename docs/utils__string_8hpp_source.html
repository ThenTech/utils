678  ATTR_MAYBE_UNUSED ATTR_NODISCARD
679  static std::string to_base64(const uint8_t *buffer, size_t length) {
680  std::string encoded;
681  const uint8_t mod = length % 3;
682  encoded.reserve(((length / 3) + (mod > 0)) * 4);
683  uint32_t temp;
684 
685  for (const uint8_t *end = --buffer + length - mod; buffer != end;) {
686  temp = (uint32_t(buffer[1]) << 16) | (uint32_t(buffer[2]) << 8) | (buffer[3]);
687  buffer += 3;
688 
689  encoded.push_back(utils::string::_base64_chars[(temp & 0x00FC0000) >> 18]);
690  encoded.push_back(utils::string::_base64_chars[(temp & 0x0003F000) >> 12]);
691  encoded.push_back(utils::string::_base64_chars[(temp & 0x00000FC0) >> 6]);
692  encoded.push_back(utils::string::_base64_chars[(temp & 0x0000003F)]);
693  }
694 
695  switch (mod) {
696  case 1:
697  temp = (uint32_t(buffer[1]) << 16);
698 
699  encoded.push_back(utils::string::_base64_chars[(temp & 0x00FC0000) >> 18]);
700  encoded.push_back(utils::string::_base64_chars[(temp & 0x0003F000) >> 12]);
701  encoded.push_back('=');
702  encoded.push_back('=');
703  break;
704  case 2:
705  temp = (uint32_t(buffer[1]) << 16) | (uint32_t(buffer[2]) << 8);
706 
707  encoded.push_back(utils::string::_base64_chars[(temp & 0x00FC0000) >> 18]);
708  encoded.push_back(utils::string::_base64_chars[(temp & 0x0003F000) >> 12]);
709  encoded.push_back(utils::string::_base64_chars[(temp & 0x00000FC0) >> 6]);
710  encoded.push_back('=');
711  break;
712  }
713 
714  return encoded;
715  }

729  /**
730  * \brief Decode the given buffer with base64.
731  *
732  * \param buffer
733  * The buffer to decode.
734  * \param length
735  * The length of the buffer to decode.
736  * \return Return an std::string containing the decoded buffer.
737  */
738  ATTR_MAYBE_UNUSED ATTR_NODISCARD
739  static std::string from_base64(const uint8_t *buffer, size_t length) {
740  if (length % 4) {
741  throw utils::exceptions::ConversionException("utils::string::base64_decode (invalid size)");
742  }
743 
744  const uint8_t *const buffer_end = buffer + length;
745  size_t decoded_length = (length / 4) * 3;
746 
747  if (length) {
748  decoded_length -= (*(buffer_end - 2) == '=') + (*(buffer_end - 1) == '=');
749  }
750 
751  std::string decoded;
752  decoded.resize(decoded_length);
753  char *iter = decoded.data() - 1;
754 
755  while (buffer < buffer_end) {
756  int32_t temp = 0;
757 
758  for (uint8_t i = 0; i < 4; ++i, ++buffer) {
759  temp <<= 6;
760  if (*buffer >= 'A' && *buffer <= 'Z') {
761  temp |= *buffer - 'A';
762  } else if (*buffer >= 'a' && *buffer <= 'z') {
763  temp |= *buffer - 'a' + 26;
764  } else if (*buffer >= '0' && *buffer <= '9') {
765  temp |= *buffer - '0' + 2 * 26;
766  } else if (*buffer == '+') {
767  temp |= 2 * 26 + 10;
768  } else if (*buffer == '/') {
769  temp |= 2 * 26 + 10 + 1;
770  } else if (*buffer == '=') {
771  switch (buffer_end - buffer) {
772  case 1:
773  *++iter = char((temp >> 16) & 0xFF);
774  *++iter = char((temp >> 8) & 0xFF);
775  return decoded;
776  case 2:
777  *++iter = char((temp >> 10) & 0xFF);
778  return decoded;
779  default:
780  throw utils::exceptions::ConversionException("utils::string::base64_decode (invalid padding)");
781  }
782  } else {
783  throw utils::exceptions::ConversionException("utils::string::base64_decode (invalid character)");
784  }
785  }
786 
787  *++iter = char((temp >> 16) & 0xFF);
788  *++iter = char((temp >> 8) & 0xFF);
789  *++iter = char((temp ) & 0xFF);
790  }
791 
792  return decoded;
793  }
794 
795  /**
796  * \brief Decode the given string with base64.
797  *
798  * \param str
799  * The string to decode.
800  * \return Return an std::string containing the decoded string.
801  */
802  ATTR_MAYBE_UNUSED ATTR_NODISCARD
803  static inline std::string from_base64(const std::string_view& str) {
804  return utils::string::from_base64(reinterpret_cast<const uint8_t*>(str.data()), str.length());
805  }
