694  * \param delim

697  * The maximum amount of splits to make.
698  * `-1` split and call on all delimiters
699  * `0` don't split: call will be made with the original string \p s
700  * `1` call will be made with the first splitted element
701  * (starting from the right), and the rest of the string (left)
702  * as second call.
703  */
704  template<typename F> ATTR_MAYBE_UNUSED
705  static void for_each_rsplitted(F&& callback,
706  const std::string_view s,
707  const utils::string::string_view delim = ',',
708  int max_splits = -1)
709  {
710  static_assert(utils::traits::is_invocable_v<F, const std::string_view>,
711  "utils::string::for_each_rsplitted: Callable function required.");
712  utils::traits::found_t found = 0ull;
713  const size_t delim_len = delim.size();
714  size_t prev_start = s.size() - delim_len;
715 
716  while (max_splits != 0 && (found = utils::string::rcontains(s, delim, prev_start))) {
717  const size_t size = prev_start - *found;
718  std::invoke<F>(std::forward<F>(callback),
719  std::string_view{ s.data() + *found + delim_len, size });
720  max_splits--;
721 
722  prev_start = *found;
723  if (*found < delim_len) break;
724  prev_start -= delim_len;
725  }
726 
727  if ((!found && prev_start+delim_len > 0) || (found && max_splits == 0)) {
728  std::invoke<F>(std::forward<F>(callback),
729  std::string_view{ s.data(), prev_start + delim_len });
730  } else if (found && *found < delim_len) {
731  std::invoke<F>(std::forward<F>(callback),
732  std::string_view{ s.data(), prev_start });
733  } else if (prev_start > delim_len) {
734  std::invoke<F>(std::forward<F>(callback), s);
735  }
736  }

927  *
928  * \param buffer
929  * The buffer to decode.
930  * \param length
931  * The length of the buffer to decode.
932  * \return Return an std::string containing the decoded buffer.
933  */
934  ATTR_MAYBE_UNUSED ATTR_NODISCARD
935  static std::string from_base64(const uint8_t *buffer, size_t length) {
936  if (HEDLEY_UNLIKELY(length % 4)) {
937  throw utils::exceptions::ConversionException("utils::string::base64_decode (invalid size)");
938  }
939 
940  const uint8_t *const buffer_end = buffer + length;
941  size_t decoded_length = (length / 4) * 3;
942 
943  if (HEDLEY_LIKELY(length)) {
944  decoded_length -= (*(buffer_end - 2) == '=') + (*(buffer_end - 1) == '=');
945  }
946 
947  std::string decoded;
948  decoded.resize(decoded_length);
949  char *iter = decoded.data() - 1;
950 
951  while (buffer < buffer_end) {
952  int32_t temp = 0;
953 
954  for (uint8_t i = 0; i < 4; ++i, ++buffer) {
955  temp <<= 6;
956  if (*buffer >= 'A' && *buffer <= 'Z') {
957  temp |= *buffer - 'A';
958  } else if (*buffer >= 'a' && *buffer <= 'z') {
959  temp |= *buffer - 'a' + 26;
960  } else if (*buffer >= '0' && *buffer <= '9') {
961  temp |= *buffer - '0' + 2 * 26;
962  } else if (*buffer == '+') {
963  temp |= 2 * 26 + 10;
964  } else if (*buffer == '/') {
965  temp |= 2 * 26 + 10 + 1;
966  } else if (*buffer == '=') {
967  switch (buffer_end - buffer) {
968  case 1:
969  *++iter = char((temp >> 16) & 0xFF);
970  *++iter = char((temp >> 8) & 0xFF);
971  return decoded;
972  case 2:
973  *++iter = char((temp >> 10) & 0xFF);
974  return decoded;
975  default:
976  throw utils::exceptions::ConversionException("utils::string::base64_decode (invalid padding)");
977  }
978  } else {
979  throw utils::exceptions::ConversionException("utils::string::base64_decode (invalid character)");
980  }
981  }
982 
983  *++iter = char((temp >> 16) & 0xFF);
984  *++iter = char((temp >> 8) & 0xFF);
985  *++iter = char((temp ) & 0xFF);
986  }
987 
988  return decoded;
989  }
