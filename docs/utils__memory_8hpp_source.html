150  old_size = new_size;

265  * This can be any type where utils::traits::is_container is true.
266  *
267  * For map-like structures, if the ::mapped_type is a pointer, it will be freed,
268  * for others if ::value_type is a pointer it will be freed.
269  *
270  * \tparam Container
271  * A container type that passes utils::traits::is_container<Container>::value.
272  * \param container
273  * The container element that holds allocated pointers.
274  */
275  template <
276  class Container,
277  typename = typename std::enable_if_t<utils::traits::is_container<Container>::value>
278  > ATTR_MAYBE_UNUSED
279  static void delete_container(Container *container) {
280  if constexpr (utils::traits::is_maplike_v<Container>) {
281  if constexpr (std::is_pointer_v<typename Container::mapped_type>)
282  {
283  #if UTILS_MEMORY_ALLOC_LOG
284  std::fprintf(stderr, "[delete_map] at 0x%p\n", container);
285  #endif
286  for (auto& [key, value] : *container) {
287  UNUSED(key);
288  utils::memory::delete_var(value);
289  }
290  }
291  } else if constexpr (std::is_pointer_v<typename Container::value_type>) {
292  #if UTILS_MEMORY_ALLOC_LOG
293  std::fprintf(stderr, "[delete_container] at 0x%p\n", container);
294  #endif
295  for (auto *value : *container) {
296  utils::memory::delete_var(value);
297  }
298  }
299 
300  utils::memory::delete_var(container);
301  }
