137  writer.put(algo::Huffman<>::DICT_HDR_ITEM_BITS,

238  /**
239  * @brief Add a single <T, Codeword> leaf to the current tree.
240  *
241  * @param pair
242  * The <T, Codeword> pair to add.
243  * Follow Codeword.word from MSB to LSB and create new Nodes on the way,
244  * ending withe a Node that has the data T.
245  */
246  void treeAddLeaf(const std::pair<T, Codeword>& pair) {
247  const size_t mask = (1 << (pair.second.len - 1)); // Mask the pair.second.len'th bit
248  size_t dirs = pair.second.word; // The directions to follow in the tree
249 
250  algo::Node<> *current = this->tree_root;
251 
252  // Grow the tree according to the dirs path, starting from MSB, except for last dir
253  for (size_t bits = pair.second.len; --bits; dirs <<= 1) {
254  // Follow the direction and create a dummy Node if none exists
255  if (dirs & mask) {
256  if (current->right == nullptr)
257  current->right = utils::memory::allocVar<algo::Node<>>(-1);
258  current = current->right;
259  } else {
260  if (current->left == nullptr)
261  current->left = utils::memory::allocVar<algo::Node<>>(-1);
262  current = current->left;
263  }
264  }
265 
266  // Create a new Node at the correct position (the last one in dirs)
267  if (dirs & mask) {
268  current->right = utils::memory::allocVar<algo::Node<>>(pair.first);
269  } else {
270  current->left = utils::memory::allocVar<algo::Node<>>(pair.first);
271  }

324  * @brief Encode bits of length sizeof(T) with Huffman encoding and
325  * write the Huffman dict and the encoded data to an outputstream.
326  *

328  * The bytestream to read from.
329  * @return Returns a new bitstream with the encoded data.
330  */
331  utils::memory::unique_t<utils::io::BitStreamWriter> encode(utils::io::BitStreamReader& reader) {
332  const size_t length = reader.get_size_bits();
333  const size_t original_length = reader.get_size();
334 
335  utils::memory::unique_t<utils::io::BitStreamWriter> writer;
336 
337  // Calculate frequencies
338  std::unordered_map<T, uint32_t> freqs;
339 
340  reader.reset();
341  while(reader.get_position() != length) {
342  const T word = T(reader.get(algo::Huffman<>::KEY_BITS));
343  freqs[word]++;
344  }
345 
346  if (freqs.empty()) {
347  // Nothing to encode?
348  writer.reset(nullptr);
349  return writer;
350  }
351 
352  // Create priority queue to sort tree with Nodes with data from frequency
353  std::priority_queue<algo::Node<>*, std::vector<algo::Node<>*>, algo::Node<>::comparator> pq;
354 
355  for (const auto& pair: freqs) {
356  pq.emplace(utils::memory::allocVar<algo::Node<>>(pair.first, pair.second));
357  // utils::Logger::Writef("%02X: %d\n", pair.first, pair.second);
358  }
359 
360  while (pq.size() > 1) {
361  // Empty out queue and build leaves, starting with lowest freq
362  // Result is a single Node with references to other Nodes in tree structure.
363  algo::Node<> *left = pq.top(); pq.pop();
364  algo::Node<> *right = pq.top(); pq.pop();
365 
366  pq.emplace(utils::memory::allocVar<algo::Node<>>(-1, left->freq + right->freq, left, right));
367  }
368 
369  // Huffman tree root
370  this->tree_root = pq.top();
371 
372  // Create dictionary by tree traversal
373  this->buildDict(this->tree_root, std::vector<bool>());
374 
375  // Create new list with dict elements sorted by bit length for saving to stream
376  std::vector<std::pair<uint8_t, algo::Codeword>> sorted_dict(this->dict.begin(), this->dict.end());
377 
378  // Sort the dictionary by value bit length
379  std::sort(sorted_dict.begin(), sorted_dict.end(), algo::Huffman<>::CodewordComparator());
380 
381  // Determine frequencies of each bit length with {bit_length: freq}
382  std::unordered_map<uint32_t, uint32_t> bit_freqs;
383  for (const auto& w : sorted_dict) {
384  bit_freqs[w.second.len]++;
385  }
386 
387  // Calculate total needed length for dict
388  size_t h_dict_total_length = (algo::Huffman<>::KEY_BITS * this->dict.size()) // Amount of bits needed for keys
389  + ((algo::Huffman<>::DICT_HDR_HAS_ITEMS_BITS + algo::Huffman<>::DICT_HDR_ITEM_BITS + algo::Huffman<>::DICT_HDR_SEQ_LENGTH_BITS)
390  * bit_freqs.size()) // Amount of bits for each header
391  + 1; // Stop bit
392  for (const auto& f : bit_freqs) {
393  h_dict_total_length += f.first * f.second; // Amount of bits for each header group
394  }
395 
396  utils::Logger::Info("[Huffman] Table overhead with %d entries: %.1f bytes.",
397  this->dict.size(), float(h_dict_total_length) / 8.0f);
398 
399  // Save the Huffman dictionary to a stream
400  writer.reset(utils::memory::allocVar<utils::io::BitStreamWriter>((h_dict_total_length + length) / 8 + 1));
401  uint32_t seq_len = 0u, bit_len = 0u;
402 
403  // Add headers for each group of same length key:val pairs
404  // and write them to the stream
405  for (const auto& w : sorted_dict) {
406  if (seq_len == 0) {
407  // New group
408  bit_len = w.second.len;
409  seq_len = bit_freqs[bit_len];
410  this->add_huffman_dict_header(seq_len, bit_len, *writer);
411  }
412 
413  writer->put(algo::Huffman<>::KEY_BITS, w.first); // Put Key
414  writer->put(bit_len, w.second.word); // Put Val
415  seq_len--;
416  }
417 
418  this->add_huffman_dict_header(0, 0, *writer); // Stop bit
419 
420  // Store length in bytes of source
421  // FIXME capped at 65k
422  writer->put(algo::Huffman<>::LEN_BITS, uint16_t(original_length));
423 
424  // Encode
425  reader.reset();
426  while(reader.get_position() < length) {
427  const T word = T(reader.get(algo::Huffman<>::KEY_BITS));
428  const auto& pair = this->dict[word];
429  writer->put(pair.len, pair.word);
430  }
431 
432  const size_t total_length = writer->get_last_byte_position();
433 
434  utils::Logger::Writef("[Huffman] Input file size: %8d bytes" + utils::Logger::CRLF, original_length);
435  utils::Logger::Writef("[Huffman] Compressed size: %8d bytes => Ratio: %.2f%%" + utils::Logger::CRLF,
436  total_length,
437  float(total_length) / original_length * 100.0f);
438 
439  if (original_length < total_length) {
440  utils::Logger::Warn("[Huffman] No extra compression achieved, reverting stream to encoded.");
441 
442  writer.reset(utils::memory::allocVar<utils::io::BitStreamWriter>(original_length));
443  writer->put_bit(0);
444 
445  reader.reset();
446  while(reader.get_position() < length) {
447  writer->put(algo::Huffman<>::KEY_BITS, reader.get(algo::Huffman<>::KEY_BITS));
448  }
449  }
450 
451  return writer;
452  }

456  * write the decoded data to an outputstream.

463  utils::memory::unique_t<utils::io::BitStreamReader> result;
464 
465  if (reader.get_size() == 0) {
466  // Nothing to decode?
467  result.reset(nullptr);
468  return result;
469  }
470 
471  this->buildTree(reader);
472 
473  const size_t raw_bits = reader.get_size_bits();
474  size_t data_bits = raw_bits - reader.get_position();
475 
476  if (this->tree_root->isLeaf()) {
477  // No tree was build => No Huffman used, just use passthrough of buffer by setting pointer
478  const size_t data_bytes = utils::bits::round_to_byte(data_bits - 8);
479 
480  auto writer = utils::memory::new_unique_var<utils::io::BitStreamWriter>(data_bytes);
481 
482  while (reader.get_position() < raw_bits-8) {
483  writer->put(algo::Huffman<>::KEY_BITS, reader.get(algo::Huffman<>::KEY_BITS));
484  }
485 
486  result.reset(utils::memory::allocVar<utils::io::BitStreamReader>(writer->get_buffer(),
487  writer->get_last_byte_position()));
488 
489  writer->set_managed(false);
490  result->set_managed(true);
491  result->set_position(writer->get_position());
492 
493  utils::Logger::Warn("[Huffman] No Huffman table present in file. Skipping decompression.");
494  } else {
495  // Get length in bytes of source
496  const size_t data_bytes = reader.get(algo::Huffman<>::LEN_BITS);
497 
498  // Consume all other data, bit by bit and traverse Huffman tree to find word
499  auto writer = utils::memory::new_unique_var<utils::io::BitStreamWriter>(data_bytes);
500 
501  while (reader.get_position() < raw_bits && writer->get_last_byte_position() < data_bytes) {
502  this->decode(reader, *writer);
503 
504  if (writer->get_position() + 16 > data_bits) {
505  // Resize buffer if decompression reaches buffer size.
506  data_bits = writer->resize() * 8u;
507  }
508  }
509 
510  const size_t original_length = reader.get_size();
511  const size_t total_length = writer->get_last_byte_position();
512 
513  result.reset(utils::memory::allocVar<utils::io::BitStreamReader>(writer->get_buffer(),
514  total_length));
515 
516  // Transfer ownership of buffer from writer to result stream
517  writer->set_managed(false);
518  result->set_managed(true);
519 
520  utils::Logger::Writef("[Huffman] Input file size: %8d bytes" + utils::Logger::CRLF, original_length);
521  utils::Logger::Writef("[Huffman] Decompressed size: %8d bytes => Ratio: %.2f%%" + utils::Logger::CRLF,
522  total_length,
523  float(total_length) / original_length * 100.0f);
524  }
525 
526  return result;
527  }
