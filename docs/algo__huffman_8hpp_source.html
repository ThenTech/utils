246  *
247  * @param pair

249  * Follow Codeword.word from MSB to LSB and create new Nodes on the way,

252  void treeAddLeaf(const KeyPair& pair) {

254  const size_t mask = utils::bits::mask_one(word_value.len); // Mask the word_value.len'th bit
255  size_t dirs = word_value.word; // The directions to follow in the tree
256 
257  algo::Node<T> *current = this->tree_root;
258 
259  // Grow the tree according to the dirs path, starting from MSB, except for last dir
260  for (size_t bits = word_value.len; --bits; dirs <<= 1) {
261  // Follow the direction and create a dummy Node if none exists
262  if (dirs & mask) {
263  if (current->right == nullptr)
264  current->right = utils::memory::new_var<algo::Node<T>>(-1);
265  current = current->right;
266  } else {
267  if (current->left == nullptr)
268  current->left = utils::memory::new_var<algo::Node<T>>(-1);
269  current = current->left;
270  }
271  }
272 
273  // Create a new Node at the correct position (the last one in dirs)
274  if (dirs & mask) {
275  current->right = utils::memory::new_var<algo::Node<T>>(key);
276  } else {
277  current->left = utils::memory::new_var<algo::Node<T>>(key);

338  const size_t original_length = reader.get_size();
339

341 
342  // Calculate frequencies
343  std::unordered_map<T, uint32_t> freqs;
344 
345  reader.reset();
346  while(reader.get_position() != length) {
347  const T word = T(reader.get(algo::Huffman<T>::KEY_BITS));
348  freqs[word]++;
349  }
350 
351  if (freqs.empty()) {
352  // Nothing to encode?
353  writer.reset(nullptr);
354  return writer;
355  }
356 
357  // Create priority queue to sort tree with Nodes with data from frequency
358  std::priority_queue<algo::Node<T>*, std::vector<algo::Node<T>*>, typename algo::Node<T>::comparator> pq;
359 
360  for (const auto& [data, freq]: freqs) {
361  pq.emplace(utils::memory::new_var<algo::Node<T>>(data, freq));
362  // utils::Logger::Writef("%02X: %d\n", data, freq);
363  }
364 
365  while (pq.size() > 1) {
366  // Empty out queue and build leaves, starting with lowest freq
367  // Result is a single Node with references to other Nodes in tree structure.
368  algo::Node<T> *left = pq.top(); pq.pop();
369  algo::Node<T> *right = pq.top(); pq.pop();
370 
371  pq.emplace(utils::memory::new_var<algo::Node<T>>(-1, left->freq + right->freq, left, right));
372  }
373 
374  // Huffman tree root
375  this->tree_root = pq.top();
376 
377  // Create dictionary by tree traversal
378  this->buildDict(this->tree_root, {});
379

381  std::vector<KeyPair> sorted_dict(this->dict.begin(), this->dict.end());
382 
383  // Sort the dictionary by value bit length
384  std::sort(sorted_dict.begin(), sorted_dict.end(), algo::Huffman<T>::CodewordComparator());
385 
386  // Determine frequencies of each bit length with {bit_length: freq}
387  std::unordered_map<uint32_t, uint32_t> bit_freqs;
388  for (const auto& [value, word] : sorted_dict) {
389  UNUSED(value);
390  bit_freqs[word.len]++;
391  }
392

394  size_t h_dict_total_length = (algo::Huffman<T>::KEY_BITS * this->dict.size()) // Amount of bits needed for keys

397  + 1; // Stop bit
398  for (const auto& [bit_length, freq] : bit_freqs) {
399  h_dict_total_length += bit_length * freq; // Amount of bits for each header group
400  }
401

410  // and write them to the stream

412  if (seq_len == 0) {

415  seq_len = bit_freqs[bit_len];

417  }
418 
419  writer->put(algo::Huffman<T>::KEY_BITS, value); // Put Key
420  writer->put(bit_len, word.word); // Put Val
421  seq_len--;
422  }
423 
424  this->add_huffman_dict_header(0, 0, *writer); // Stop bit
425 
426  // Store length in bytes of source
427  // FIXME capped at 65k due to LEN_BITS being *only* 16
428  // Maybe 2 additional bits to indicate length of [0b00: 16, 0b01: 32: 0b10: 48, 0b11: 64]?
429  writer->put(algo::Huffman<T>::LEN_BITS, uint16_t(original_length));
430 
431  // Encode
432  reader.reset();
433  while (reader.get_position() < length) {
434  const T word = T(reader.get(algo::Huffman<T>::KEY_BITS));
435  const auto& pair = this->dict[word];
436  writer->put(pair.len, pair.word);
437  }

440 
441  utils::Logger::Info("[Huffman] Input file size: %8d bytes", original_length);
442  utils::Logger::Info("[Huffman] Compressed size: %8d bytes => Ratio: %.2f%%",
443  total_length,
444  float(total_length) / original_length * 100.0f);
445

447  utils::Logger::Warn("[Huffman] No extra compression achieved, reverting stream to encoded.");
448

450  writer->put_bit(0);

454  writer->put(algo::Huffman<T>::KEY_BITS, reader.get(algo::Huffman<T>::KEY_BITS));

524  writer->set_managed(false);
525  result->set_managed(true);

528  utils::Logger::Info("[Huffman] Decompressed size: %8d bytes => Ratio: %.2f%%",

531  }
