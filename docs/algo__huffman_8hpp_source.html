131  * Keys always use KEY_BITS as length, and values use bit_length, which is different for each group.

139  writer.put(algo::Huffman<T>::DICT_HDR_HAS_ITEMS_BITS, 1);

141  utils::bits::select_lsb<uint32_t>(length, algo::Huffman<T>::DICT_HDR_SEQ_LENGTH_BITS));
142 
143  // 4 bits for bit length of every dict item

145  utils::bits::select_lsb<uint32_t>(bit_length, algo::Huffman<T>::DICT_HDR_ITEM_BITS));

244  /**
245  * @brief Add a single <T, Codeword> leaf to the current tree.
246  *
247  * @param pair
248  * The <T, Codeword> pair to add.
249  * Follow Codeword.word from MSB to LSB and create new Nodes on the way,
250  * ending withe a Node that has the data T.
251  */
252  void treeAddLeaf(const KeyPair& pair) {
253  const size_t mask = utils::bits::mask_one(pair.second.len); // Mask the pair.second.len'th bit
254  size_t dirs = pair.second.word; // The directions to follow in the tree
255 
256  algo::Node<T> *current = this->tree_root;
257 
258  // Grow the tree according to the dirs path, starting from MSB, except for last dir
259  for (size_t bits = pair.second.len; --bits; dirs <<= 1) {
260  // Follow the direction and create a dummy Node if none exists
261  if (dirs & mask) {
262  if (current->right == nullptr)
263  current->right = utils::memory::new_var<algo::Node<T>>(-1);
264  current = current->right;
265  } else {
266  if (current->left == nullptr)
267  current->left = utils::memory::new_var<algo::Node<T>>(-1);
268  current = current->left;
269  }
270  }
271 
272  // Create a new Node at the correct position (the last one in dirs)
273  if (dirs & mask) {
274  current->right = utils::memory::new_var<algo::Node<T>>(pair.first);
275  } else {
276  current->left = utils::memory::new_var<algo::Node<T>>(pair.first);
277  }
278  }

328  * @brief Encode bits of length sizeof(T) with Huffman encoding and
329  * write the Huffman dict and the encoded data to an outputstream.
330  *
331  * @param reader
332  * The bytestream to read from.
333  * @return Returns a new bitstream with the encoded data.
334  */
335  utils::memory::unique_t<utils::io::BitStreamWriter> encode(utils::io::BitStreamReader& reader) {
336  const size_t length = reader.get_size_bits();
337  const size_t original_length = reader.get_size();
338 
339  utils::memory::unique_t<utils::io::BitStreamWriter> writer;
340 
341  // Calculate frequencies
342  std::unordered_map<T, uint32_t> freqs;
343 
344  reader.reset();
345  while(reader.get_position() != length) {
346  const T word = T(reader.get(algo::Huffman<T>::KEY_BITS));
347  freqs[word]++;
348  }
349 
350  if (freqs.empty()) {
351  // Nothing to encode?
352  writer.reset(nullptr);
353  return writer;
354  }
355 
356  // Create priority queue to sort tree with Nodes with data from frequency
357  std::priority_queue<algo::Node<T>*, std::vector<algo::Node<T>*>, typename algo::Node<T>::comparator> pq;
358 
359  for (const auto& [data, freq]: freqs) {
360  pq.emplace(utils::memory::new_var<algo::Node<T>>(data, freq));
361  // utils::Logger::Writef("%02X: %d\n", data, freq);
362  }
363 
364  while (pq.size() > 1) {
365  // Empty out queue and build leaves, starting with lowest freq
366  // Result is a single Node with references to other Nodes in tree structure.
367  algo::Node<T> *left = pq.top(); pq.pop();
368  algo::Node<T> *right = pq.top(); pq.pop();
369 
370  pq.emplace(utils::memory::new_var<algo::Node<T>>(-1, left->freq + right->freq, left, right));
371  }
372 
373  // Huffman tree root
374  this->tree_root = pq.top();
375 
376  // Create dictionary by tree traversal
377  this->buildDict(this->tree_root, {});
378 
379  // Create new list with dict elements sorted by bit length for saving to stream
380  std::vector<KeyPair> sorted_dict(this->dict.begin(), this->dict.end());
381 
382  // Sort the dictionary by value bit length
383  std::sort(sorted_dict.begin(), sorted_dict.end(), algo::Huffman<T>::CodewordComparator());
384 
385  // Determine frequencies of each bit length with {bit_length: freq}
386  std::unordered_map<uint32_t, uint32_t> bit_freqs;
387  for (const auto& [value, word] : sorted_dict) {
388  UNUSED(value);
389  bit_freqs[word.len]++;
390  }
391 
392  // Calculate total needed length for dict
393  size_t h_dict_total_length = (algo::Huffman<T>::KEY_BITS * this->dict.size()) // Amount of bits needed for keys
394  + ((algo::Huffman<T>::DICT_HDR_HAS_ITEMS_BITS + algo::Huffman<T>::DICT_HDR_ITEM_BITS + algo::Huffman<T>::DICT_HDR_SEQ_LENGTH_BITS)
395  * bit_freqs.size()) // Amount of bits for each header
396  + 1; // Stop bit
397  for (const auto& [bit_length, freq] : bit_freqs) {
398  h_dict_total_length += bit_length * freq; // Amount of bits for each header group
399  }
400 
401  utils::Logger::Info("[Huffman] Table overhead with %d entries: %.1f bytes.",
402  this->dict.size(), float(h_dict_total_length) / 8.0f);
403 
404  // Save the Huffman dictionary to a stream
405  writer.reset(utils::memory::new_var<utils::io::BitStreamWriter>((h_dict_total_length + length) / 8 + 1));
406  uint32_t seq_len = 0u, bit_len = 0u;
407 
408  // Add headers for each group of same length key:val pairs
409  // and write them to the stream
410  for (const auto& [value, word] : sorted_dict) {
411  if (seq_len == 0) {
412  // New group
413  bit_len = word.len;
414  seq_len = bit_freqs[bit_len];
415  this->add_huffman_dict_header(seq_len, bit_len, *writer);
416  }
417 
418  writer->put(algo::Huffman<T>::KEY_BITS, value); // Put Key
419  writer->put(bit_len, word.word); // Put Val
420  seq_len--;
421  }
422 
423  this->add_huffman_dict_header(0, 0, *writer); // Stop bit
424 
425  // Store length in bytes of source
426  // FIXME capped at 65k due to LEN_BITS being *only* 16
427  // Maybe 2 additional bits to indicate length of [0b00: 16, 0b01: 32: 0b10: 48, 0b11: 64]?
428  writer->put(algo::Huffman<T>::LEN_BITS, uint16_t(original_length));
429 
430  // Encode
431  reader.reset();
432  while(reader.get_position() < length) {
433  const T word = T(reader.get(algo::Huffman<T>::KEY_BITS));
434  const auto& pair = this->dict[word];
435  writer->put(pair.len, pair.word);
436  }
437 
438  const size_t total_length = writer->get_last_byte_position();
439 
440  utils::Logger::Info("[Huffman] Input file size: %8d bytes", original_length);
441  utils::Logger::Info("[Huffman] Compressed size: %8d bytes => Ratio: %.2f%%",
442  total_length,
443  float(total_length) / original_length * 100.0f);
444 
445  if (original_length < total_length) {
446  utils::Logger::Warn("[Huffman] No extra compression achieved, reverting stream to encoded.");
447 
448  writer.reset(utils::memory::new_var<utils::io::BitStreamWriter>(original_length));
449  writer->put_bit(0);
450 
451  reader.reset();
452  while(reader.get_position() < length) {
453  writer->put(algo::Huffman<T>::KEY_BITS, reader.get(algo::Huffman<T>::KEY_BITS));
454  }
455  }
456 
457  return writer;
458  }

462  * write the decoded data to an outputstream.

469  utils::memory::unique_t<utils::io::BitStreamReader> result;
470 
471  if (reader.get_size() == 0) {
472  // Nothing to decode?
473  result.reset(nullptr);
474  return result;
475  }
476 
477  this->buildTree(reader);
478 
479  const size_t raw_bits = reader.get_size_bits();
480  size_t data_bits = raw_bits - reader.get_position();
481 
482  if (this->tree_root->isLeaf()) {
483  // No tree was build => No Huffman used, just use passthrough of buffer by setting pointer
484  const size_t data_bytes = utils::bits::round_to_byte(data_bits - 8);
485 
486  auto writer = utils::memory::new_unique_var<utils::io::BitStreamWriter>(data_bytes);
487 
488  while (reader.get_position() < raw_bits-8) {
489  writer->put(algo::Huffman<T>::KEY_BITS, reader.get(algo::Huffman<T>::KEY_BITS));
490  }
491 
492  result.reset(utils::memory::new_var<utils::io::BitStreamReader>(writer->get_buffer(),
493  writer->get_last_byte_position()));
494 
495  writer->set_managed(false);
496  result->set_managed(true);
497  result->set_position(writer->get_position());
498 
499  utils::Logger::Warn("[Huffman] No Huffman table present in file. Skipping decompression.");
500  } else {
501  // Get length in bytes of source
502  const size_t data_bytes = reader.get(algo::Huffman<T>::LEN_BITS);
503 
504  // Consume all other data, bit by bit and traverse Huffman tree to find word
505  auto writer = utils::memory::new_unique_var<utils::io::BitStreamWriter>(data_bytes);
506 
507  while (reader.get_position() < raw_bits && writer->get_last_byte_position() < data_bytes) {
508  this->decode(reader, *writer);
509 
510  if (writer->get_position() + 16 > data_bits) {
511  // Resize buffer if decompression reaches buffer size.
512  data_bits = writer->resize() * 8u;
513  }
514  }
515 
516  const size_t original_length = reader.get_size();
517  const size_t total_length = writer->get_last_byte_position();
518 
519  result.reset(utils::memory::new_var<utils::io::BitStreamReader>(writer->get_buffer(),
520  total_length));
521 
522  // Transfer ownership of buffer from writer to result stream
523  writer->set_managed(false);
524  result->set_managed(true);
525 
526  utils::Logger::Info("[Huffman] Input file size: %8d bytes", original_length);
527  utils::Logger::Info("[Huffman] Decompressed size: %8d bytes => Ratio: %.2f%%",
528  total_length,
529  float(total_length) / original_length * 100.0f);
530  }
531 
532  return result;
533  }
