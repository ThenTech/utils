48  void convert_keys(const uint64_t *init_key, const size_t length) {
49  if (length == rounds) {
50  std::copy(init_key, init_key + length, this->keys.data());
51  } else {
52  static const utils::CRC::Table<uint64_t, 64> table(CRC_64);
53 
54  if (length < rounds) {
55  // Create extra values to fill rounds
56  std::copy(init_key, init_key + length, this->keys.data());
57 
58  const uint64_t crc = utils::CRC::Calculate(init_key, length, table);
59  auto&& rnd_seed = std::seed_seq{crc, default_keys[0], default_keys[1], default_keys[2]};
60  constexpr uint64_t max = static_cast<uint64_t>(-1);
61 
62  utils::random::RandomLocal r;
63  r.seed(rnd_seed);
64 
65  for (size_t i = length; i < rounds; ++i) {
66  this->keys[i] = r.get(uint64_t{}, max);
67  }
68  } else {
69  // Consume extra values to fit rounds
70  std::copy(init_key, init_key + rounds, this->keys.data());
71  this->keys.back() = utils::CRC::Calculate(init_key + rounds - 1, length - rounds + 1, table);
72  }

76  uint64_t encrypt(uint32_t left, uint32_t right) const {
77  Hasher h{};
78 
79  for (size_t round = 0; round < rounds; ++round) {
80  const uint32_t prev_right = right;
81  right = left;
82  left = h(left, this->keys[round]) ^ prev_right;
