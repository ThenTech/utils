92  /**
93  * brief Get the colour from the given hex string.
94  *
95  * param hexstr
96  * Must be in hexadecimal representation,
97  * prefixed with either `0x`, `0X` or `#`.
98  * return Returns a Colour instance containing the colour in \p hexstr.
99  */
100  static inline Colour from_hex(const char* hexstr) {
101  const auto hexcast = utils::misc::try_lexical_cast<uint32_t>(hexstr);
102  const uint32_t hexnum = hexcast.value_or(0);
103 
104  return Colour{
105  static_cast<double>((hexnum >> 16) & 0xFF) / 255.0,
106  static_cast<double>((hexnum >> 8) & 0xFF) / 255.0,
107  static_cast<double>((hexnum ) & 0xFF) / 255.0
108  };
109  }

159  ATTR_MAYBE_UNUSED
160  static Colour GetHeatColor(double x) {
161  x = std::clamp(x, 0.0, 1.0);
162 
163  static constexpr Colour data[] = {
164  { 0.0, 0.0, 1.0 },
165  { 0.0, 1.0, 1.0 },
166  { 0.0, 1.0, 0.0 },
167  { 1.0, 1.0, 0.0 },
168  { 1.0, 0.0, 0.0 }
169  };
170 
171  const double a = x * ((sizeof(data) / sizeof(Colour)) - 1);
172  const double t = a - std::floor(a);
173  const Colour c0 = data[static_cast<size_t>(std::floor(a))];
174  const Colour c1 = data[static_cast<size_t>(std::ceil (a))];
175 
176  return (1.0 - t) * c0 + t * c1;
177  }

203  ATTR_MAYBE_UNUSED
204  static Colour GetHotColor(double x) {
205  x = std::clamp(x, 0.0, 1.0);
206 
207  static constexpr Colour r{ 1.0, 0.0, 0.0 };
208  static constexpr Colour g{ 0.0, 1.0, 0.0 };
209  static constexpr Colour b{ 0.0, 0.0, 1.0 };
210 
211  if (x < 0.4) {
212  const double t = x / 0.4;
213  return t * r;
214  } else if (x < 0.8) {
215  const double t = (x - 0.4) / (0.8 - 0.4);
216  return r + t * g;
217  } else {
218  const double t = (x - 0.8) / (1.0 - 0.8);
219  return r + g + t * b;
220  }
221  }
