362  /**
363  * \brief Dump the given data to ascii output and hex view.
364  *
365  * \param stream
366  * The stream to write to.
367  * \param data
368  * The data to dump as raw `void*`, will be cast to `char*`.
369  * \param length
370  * Length of \p data.
371  * \param width
372  * Width for dump format, equal to amount of bytes on each line.
373  */
374  template<typename TChar, typename TCharTraits> ATTR_MAYBE_UNUSED
375  static void hexdump(std::basic_ostream<TChar, TCharTraits>& stream,
376  const void* data, size_t length,
377  size_t width = 16)
378  {
379  if (data == nullptr)
380  return;
381 
382  const char* const start = static_cast<const char*>(data);
383  const char* const end = start + length;
384  const char* line = start;
385  const auto locale = stream.getloc();
386 
387  while (line != end) {
388  const size_t lineLength = std::min(width, static_cast<std::size_t>(end - line));
389 
390  stream << utils::string::format("%04X : ", line - start);
391 
392  for (const char* next = line; next != line + lineLength; ++next) {
393  stream << (!std::isprint(*next, locale) ? '.' : *next);
394  }
395 
396  stream << std::string(width - lineLength, ' ');
397 
398  for (const char* next = line; next != line + lineLength; ++next) {
399  stream << utils::string::format(" %02X", (*next) & 0xFF);
400  }
401 
402  stream << std::endl;
403  line += lineLength;
404  }
405  }
