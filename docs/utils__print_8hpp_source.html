345  * **Uses __abi::__cxa_demangle__ which is part of <cxxabi.h> included in all GCC compilers.**
346  *
347  * If GCC is not used, type2name will revert to typeid(o).name() instead.
348  *

351  * \param o

354  * A list of strings to filter (remove) from the output type name.
355  * \return
356  * Returns the class name of o.
357  */

360  typename = typename std::enable_if_t<(std::is_convertible_v<Strings const&, std::string_view> && ...)>

369  #else
370  std::string s(typeid(o).name());

381  /**
382  * \brief Dump the given data to ascii output and hex view.
383  *
384  * \param stream
385  * The stream to write to.
386  * \param data
387  * The data to dump as raw `void*`, will be cast to `char*`.
388  * \param length
389  * Length of \p data.
390  * \param width
391  * Width for dump format, equal to amount of bytes on each line.
392  */
393  template<typename TChar, typename TCharTraits> ATTR_MAYBE_UNUSED
394  static void hexdump(std::basic_ostream<TChar, TCharTraits>& stream,
395  const void* data, size_t length,
396  size_t width = 16)
397  {
398  if (data == nullptr)
399  return;
400 
401  const char* const start = static_cast<const char*>(data);
402  const char* const end = start + length;
403  const char* line = start;
404  const auto locale = stream.getloc();
405 
406  while (line != end) {
407  const size_t lineLength = std::min(width, static_cast<std::size_t>(end - line));
408 
409  stream << utils::string::format("%04X : ", line - start);
410 
411  for (const char* next = line; next != line + lineLength; ++next) {
412  stream << (!std::isprint(*next, locale) ? '.' : *next);
413  }
414 
415  stream << std::string(width - lineLength, ' ');
416 
417  for (const char* next = line; next != line + lineLength; ++next) {
418  stream << utils::string::format(" %02X", (*next) & 0xFF);
419  }
420 
421  stream << std::endl;
422  line += lineLength;
423  }
424  }
425 }

526  /**
527  * \brief Generic operator <<
528  *
529  * if T == is_pointer<T>
530  * => prints %p format if void*, "nullptr" if nullptr or else dereference
531  * if T == __gnu_cxx::__normal_iterator<T, Container>
532  * or another iterator type
533  * => prints dereference
534  * if T == is_array<T>
535  * => prints with array_wrapper
536  * if T == is_container<T>
537  * => prints print_container_helpter<T> using default delimiters
538  * else prints value itself.
539  * => prints <Object type2name(value)>
540  * or ASSERT(false)
541  *
542  * \param stream
543  * The stream to write to.
544  * \param value
545  * The possible pointer/iterator/array/container value.
546  * \return Returns the stream reference.
547  */
548  template<typename T, typename TChar, typename TCharTraits>
549  auto& operator<<(
550  std::basic_ostream<TChar, TCharTraits>& stream,
551  const T& value)
552  {
553  if constexpr (std::is_pointer_v<T>) {
554  if constexpr (std::is_void_v<std::remove_pointer_t<T>>) {
555  stream << (value == nullptr ? "nullptr" : utils::string::format("%p", value));
556  } else {
557  stream << *value;
558  }
559  } else if constexpr (utils::traits::is_iterator_v<T>) {
560  stream << *value;
561  } else if constexpr (std::is_array_v<T>) {
562  stream << utils::print::array_wrapper(value);
563  } else if constexpr (utils::traits::is_container<T>::value) {
564  (utils::print::print_container_helper<T, TChar, TCharTraits>(value))(stream);
565  } else {
566  stream << "<Object " << utils::print::type2name(value) << ">";
567  // ASSERT(false && "Warning: Unprintable value");
568  }
569 
570  return stream;
571  }
