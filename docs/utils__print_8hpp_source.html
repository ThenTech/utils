406  /**
407  * \brief Dump the given data to ascii output and hex view.
408  *
409  * \param stream
410  * The stream to write to.
411  * \param data
412  * The data to dump as raw `void*`, will be cast to `char*`.
413  * \param length
414  * Length of \p data.
415  * \param width
416  * Width for dump format, equal to amount of bytes on each line.
417  */
418  template<typename TChar, typename TCharTraits> ATTR_MAYBE_UNUSED
419  static void hexdump(std::basic_ostream<TChar, TCharTraits>& stream,
420  const void* data, size_t length,
421  size_t width = 16)
422  {
423  if (HEDLEY_UNLIKELY(data == nullptr)) {
424  return;
425  }
426 
427  const char* const start = static_cast<const char*>(data);
428  const char* const end = start + length;
429  const char* line = start;
430  const auto locale = stream.getloc();
431 
432  while (line != end) {
433  const size_t lineLength = std::min(width, static_cast<std::size_t>(end - line));
434 
435  stream << utils::string::format("%04X : ", line - start);
436 
437  for (const char* next = line; next != line + lineLength; ++next) {
438  stream << (!std::isprint(*next, locale) ? '.' : *next);
439  }
440 
441  stream << std::string(width - lineLength, ' ');
442 
443  for (const char* next = line; next != line + lineLength; ++next) {
444  stream << utils::string::format(" %02X", (*next) & 0xFF);
445  }
446 
447  stream << std::endl;
448  line += lineLength;
449  }

495  this->iteration = std::min(this->iteration, this->max_iteration);
496  this->progress = double(this->iteration) / this->max_iteration;

502  template<typename TChar, typename TCharTraits>
503  void print(std::basic_ostream<TChar, TCharTraits>& stream = std::cout) const {
504  if (HEDLEY_UNLIKELY(this->finished)) {
505  return;
506  }
507 
508  if (HEDLEY_LIKELY(this->update_on_step_only && !this->should_redraw)) {
509  return;
510  }
511 
512  stream << utils::os::Console::CLLINE << '\r'
513  << this->prefix;
514 
515  if (HEDLEY_LIKELY(this->with_bar)) {
516  size_t filled = 0;
517  stream << "[";
518  while (++filled < this->step) stream << FILL_CHAR;
519  while (filled++ < this->width) stream << EMPTY_CHAR;
520  stream << "]";
521  }
522 
523  if (HEDLEY_LIKELY(this->with_percentage)) {
524  stream << utils::string::format(" %5.1f%%", this->progress * 100.0f);
525  }
526 
527  if (HEDLEY_LIKELY(this->with_elapsed || this->with_remaining || this->with_per_second)) {
528  const auto elapsed_ms = utils::time::Timer::time_ms::duration(this->start);
529  const double ticks_per_second = double(this->iteration) / elapsed_ms * 1e3;
530 
531  if (HEDLEY_LIKELY(this->with_elapsed)) {
532  stream << utils::time::Timestamp(std::time_t(elapsed_ms / 1000.0), " %M:%S")
533  << utils::string::format(".%1d", (int(elapsed_ms) % 1000) / 100);
534  }
535 
536  if (HEDLEY_UNLIKELY(this->with_remaining)) {
537  const double remaining_ticks = this->max_iteration - this->iteration;
538  const double total_remaining_s = remaining_ticks / ticks_per_second;
539  stream << utils::string::format(" %3.1fs left", total_remaining_s);
540  }
541 
542  if (HEDLEY_UNLIKELY(this->with_per_second)) {
543  stream << utils::string::format(" (%1.0f/s)", ticks_per_second);
544  }
545  }
546 
547  if (HEDLEY_UNLIKELY(this->finished || (this->iteration >= this->max_iteration))) {
548  stream << "\r\n";
549  }
550 
551  stream << std::flush;
552  }

824  /**
825  * \brief Generic operator <<
826  *
827  * if T == is_pointer<T>
828  * => prints %p format if void*, "nullptr" if nullptr or else dereference
829  * if T == __gnu_cxx::__normal_iterator<T, Container>
830  * or another iterator type
831  * => prints dereference
832  * if T == is_array<T>
833  * => prints with array_wrapper
834  * if T == is_container<T>
835  * => prints print_container_helpter<T> using default delimiters
836  * else prints value itself.
837  * => prints <Object type2name(value)>
838  * or ASSERT(false)
839  *
840  * \param stream
841  * The stream to write to.
842  * \param value
843  * The possible pointer/iterator/array/container value.
844  * \return Returns the stream reference.
845  */
846  template<typename T, typename TChar, typename TCharTraits>
847  auto& operator<<(
848  std::basic_ostream<TChar, TCharTraits>& stream,
849  const T& value)
850  {
851  if constexpr (std::is_pointer_v<T>) {
852  if constexpr (std::is_void_v<std::remove_pointer_t<T>>) {
853  stream << (value == nullptr ? "nullptr" : utils::string::format("%p", value));
854  } else {
855  stream << *value;
856  }
857  } else if constexpr (utils::traits::is_iterator_v<T>) {
858  stream << *value;
859  } else if constexpr (std::is_array_v<T>) {
860  stream << utils::print::array_wrapper(value);
861  } else if constexpr (utils::traits::is_container<T>::value) {
862  (utils::print::print_container_helper<T, TChar, TCharTraits>(value))(stream);
863  } else {
864  stream << "<Object " << utils::print::type2name(value) << ">";
865  // ASSERT(false && "Warning: Unprintable value");
866  }
867 
868  return stream;
869  }
