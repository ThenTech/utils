116  * This is used for overcoming disturbances, and ensuring that the

343  /**
344  * \brief Calculate the PID value needed to hit the target setpoint.
345  * Automatically re-calculates the output at each call.
346  *
347  * \param actual
348  * The actual value of the system.
349  * \param setpoint
350  * The target value of the system.
351  * \return Calculated output value for driving the actual to the target.
352  */
353  double getOutput(double actual, double setpoint) {
354  this->setpoint = setpoint;
355 
356  // Ramp the setpoint used for calculations if user has opted to do so
357  if (this->setpointRange != 0.0) {
358  setpoint = std::clamp(setpoint,
359  actual - this->setpointRange,
360  actual + this->setpointRange);
361  }
362 
363  // Do the simple parts of the calculations
364  const double error = setpoint - actual;
365 
366  // Calculate P term
367  const double Poutput = this->P * error;
368 
369  // Calculate F output. Notice, this->depends only on the setpoint, and not the error.
370  const double Foutput = this->F * setpoint;
371 
372  // If this->is our first time running this, we don't actually _have_
373  // a previous input or output.
374  // For sensor, sanely assume it was exactly where it is now.
375  // For last output, we can assume it's the current time-independent outputs.
376  if (this->firstRun) {
377  this->lastActual = actual;
378  this->lastOutput = Poutput + Foutput;
379  this->firstRun = false;
380  }
381 
382  // Interim checks
383  const bool min_out_nequals_max = utils::math::epsilon_equals(this->minOutput, this->maxOutput, 1e-6);
384  const bool outrampraqte_not_zero = this->outputRampRate != 0.0;
385  const bool max_iout_not_zero = this->maxIOutput != 0.0;
386  const double lastout_min_ramp = this->lastOutput - this->outputRampRate;
387  const double lastout_pls_ramp = this->lastOutput + this->outputRampRate;
388 
389  // Calculate D Term
390  // Note, this is negative. This actually "slows" the system if it's doing
391  // the correct thing, and small values helps prevent output spikes and overshoot
392  const double Doutput = -this->D * (actual - this->lastActual);
393  this->lastActual = actual;
394 
395  // The Iterm is more complex. There's several things to factor in to make
396  // it easier to deal with.
397  // 1. maxIoutput restricts the amount of output contributed by the Iterm.
398  // 2. prevent windup by not increasing errorSum if we're already running
399  // against our max Ioutput
400  // 3. prevent windup by not increasing errorSum if output is output=maxOutput
401  double Ioutput = this->I * this->errorSum;
402  if (max_iout_not_zero) {
403  Ioutput = std::clamp(Ioutput, -this->maxIOutput, this->maxIOutput);
404  }
405 
406  //And, finally, we can just add the terms up
407  double output = Foutput + Poutput + Ioutput + Doutput;
408 
409  //Figure out what we're doing with the error.
410  if (min_out_nequals_max && !utils::math::within(output, this->minOutput, this->maxOutput)) {
411  // reset the error sum to a sane level
412  // Setting to current error ensures a smooth transition when the P term
413  // decreases enough for the I term to start acting upon the controller
414  // From that point the I term will build up as would be expected
415  this->errorSum = error;
416  } else if ( outrampraqte_not_zero
417  && !utils::math::within(output, lastout_min_ramp, lastout_pls_ramp))
418  {
419  this->errorSum = error;
420  } else if (max_iout_not_zero) {
421  // In addition to output limiting directly, we also want to prevent I term
422  // buildup, so restrict the error directly
423  this->errorSum = std::clamp(this->errorSum + error, -this->maxError, this->maxError);
424  } else {
425  this->errorSum += error;
426  }
427 
428  //Restrict output to our specified output and ramp limits
429  if (outrampraqte_not_zero) {
430  output = std::clamp(output, lastout_min_ramp, lastout_pls_ramp);
431  }
432 
433  if (min_out_nequals_max) {
434  output = std::clamp(output, this->minOutput, this->maxOutput);
435  }
436 
437  if (this->outputFilter != 0.0) {
438  output = utils::math::mix(output, this->lastOutput, this->outputFilter);
439  }
440 
441  this->lastOutput = output;
442  return output;
443  }
