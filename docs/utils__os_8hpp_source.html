114  /**
115  * \brief Process the given commands \p cmd and write the appropriate
116  * virtual terminal codes to the given stream.
117  *
118  * For Windows, call EnableVirtualConsole() once (standard output handle only).
119  *
120  * \param cmd
121  * The commands to process.
122  * \param out
123  * The stream to write the virtual codes to. (default: std::cout)
124  */
125  template<typename TChar, typename TCharTraits> ATTR_MAYBE_UNUSED
126  static void Command(const command_t cmd, std::basic_ostream<TChar, TCharTraits>& out = std::cout) {
127  if (HEDLEY_UNLIKELY(!utils::os::internal::_virtual_console_enabled))
128  return;
129 
130  #define BASE_ "\033["
131  out.flush();
132 
133  if (cmd & Console::RESET) {
134  out << BASE_ "0m";
135  }
136 
137  if (cmd & Console::CLS) {
138  out << BASE_ "2J";
139  } else if (cmd & Console::CLLINE) {
140  out << BASE_ "2K";
141  }
142 
143  if (cmd & Console::CURSOR) {
144  out << BASE_ "H";
145  }
146 
147  if (cmd & Console::BOLD) {
148  out << BASE_ "1m";
149  }
150 
151  if (cmd & Console::UNDERLINE) {
152  out << BASE_ "4m";
153  }
154 
155  if (cmd & Console::ITALIC) {
156  out << BASE_ "3m";
157  }
158 
159  if (cmd >= Console::BLACK) {
160  out << BASE_;
161  // A colour will be set
162  if (cmd & Console::BG) {
163  out << "4"; // Background
164  } else {
165  out << "3"; // Foreground
166  }
167 
168  out << (utils::bits::msb(uint64_t(cmd) / uint64_t(Console::BLACK)) - 1);
169 
170  if (cmd & Console::BRIGHT) {
171  out << ";1";
172  }
173 
174  out << "m";
175  }
176  }

199  * \brief Enable virtual commands processing for the default output handle
200  * on Windows. Unix is supported by default.
201  */
202  ATTR_MAYBE_UNUSED
203  static void EnableVirtualConsole() {
204  int error = 0;
205 
206  #ifdef UTILS_OS_ENABLE_VIRTUAL_TERMINAL
207  // Set-up Windows terminal
208  do {
209  HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
210  if (hOut == INVALID_HANDLE_VALUE) {
211  error = int(GetLastError());
212  break;
213  }
214 
215  DWORD dwMode = 0;
216  if (!GetConsoleMode(hOut, &dwMode)) {
217  error = int(GetLastError());
218  break;
219  }
220 
221  dwMode |= UTILS_OS_ENABLE_VIRTUAL_TERMINAL;
222  if (!SetConsoleMode(hOut, dwMode)) {
223  error = int(GetLastError());
224  break;
225  }
226  } while(false);
227  #else
228  // Unix is already good.
229  #endif
230 
231  if (error) {
232  std::cerr << "[utils::os::EnableVirtualConsole] Error: " << error << std::endl;
233  } else {
234  utils::os::internal::_virtual_console_enabled = true;
235  }
236  }
