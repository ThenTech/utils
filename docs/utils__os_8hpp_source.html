119  /**
120  * \brief Process the given commands \p cmd and write the appropriate
121  * virtual terminal codes to the given stream.
122  *
123  * For Windows, call EnableVirtualConsole() once (standard output handle only).
124  *
125  * \param cmd
126  * The commands to process.
127  * \param out
128  * The stream to write the virtual codes to. (default: std::cout)
129  */
130  template<typename TChar, typename TCharTraits> ATTR_MAYBE_UNUSED
131  static void Command(const command_t cmd, std::basic_ostream<TChar, TCharTraits>& out = std::cout) {
132  if (HEDLEY_UNLIKELY(!utils::os::internal::_virtual_console_enabled))
133  return;
134 
135  #define BASE_ "\033["
136  out.flush();
137 
138  if (cmd & Console::RESET) {
139  out << BASE_ "0m";
140  }
141 
142  if (cmd & Console::CLS) {
143  out << BASE_ "2J";
144  } else if (cmd & Console::CLLINE) {
145  out << BASE_ "2K";
146  }
147 
148  if (cmd & Console::CURSOR) {
149  out << BASE_ "H";
150  }
151 
152  if (cmd & Console::BOLD) {
153  out << BASE_ "1m";
154  }
155 
156  if (cmd & Console::UNDERLINE) {
157  out << BASE_ "4m";
158  }
159 
160  if (cmd & Console::ITALIC) {
161  out << BASE_ "3m";
162  }
163 
164  if (cmd >= Console::BLACK) {
165  out << BASE_;
166  // A colour will be set
167  if (cmd & Console::BG) {
168  out << "4"; // Background
169  } else {
170  out << "3"; // Foreground
171  }
172 
173  out << (utils::bits::msb(uint64_t(cmd) / uint64_t(Console::BLACK)) - 1);
174 
175  if (cmd & Console::BRIGHT) {
176  out << ";1";
177  }
178 
179  out << "m";
180  }
181  }
