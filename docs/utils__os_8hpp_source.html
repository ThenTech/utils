144

146  cmd_str += BASE_ "3m";
147  }
148 
149  if (cmd >= Console::BLACK) {
150  // A colour will be set
151  if (cmd & Console::BG) {
152  cmd_str += BASE_ "4";
153  } else {
154  // Foreground
155  cmd_str += BASE_ "3";
156  }
157

159 
160  if (cmd & Console::BRIGHT) {
161  cmd_str += ";1";
162  }
163 
164  cmd_str += "m";
165  }
166 
167  out << cmd_str;
